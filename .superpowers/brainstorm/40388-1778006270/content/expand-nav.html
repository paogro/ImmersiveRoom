zurueckZuAncestor — Navigationslogik

Pfad-Rekonstruktion beim Zurücknavigieren
// Vollständiger Pfad = pfad + [fokusThema]
let vollPfad = pfad + [fokusThema!]
let idx = vollPfad.firstIndex { $0.id == target.id }!

// Neuer Zustand:
pfad = Array(vollPfad[0..<idx]) // alles vor dem Ziel
fokusThema = target
childrenThemen = reload(target.id)
aktuelleThemen = reload(target.parentId ?? rootId)
